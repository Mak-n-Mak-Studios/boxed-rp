383 private static float CalculateTurnFactor( float direction, float speed )
384 {
385 var turnFactor = MathF.Min( speed / 500.0f, 1 );
386 var yawSpeedFactor = 1.0f - (speed / 1000.0f).Clamp( 0, 0.6f );
387
388 return direction * turnFactor * yawSpeedFactor;

391 private static Vector3 VelocityDamping( Vector3 velocity, Rotation rotation, Vector3 damping, float dt )
392 {
393 var localVelocity = rotation.Inverse * velocity;
394 var dampingPow = new Vector3( MathF.Pow( 1.0f - damping.x, dt ), MathF.Pow( 1.0f - damping.y, dt ), MathF.Pow( 1.0f - damping.z, dt ) );
395 return rotation * (localVelocity * dampingPow);

399 {

416
417 backWheels =
418 backLeft.Raycast( length - tiltAmount - leanAmount, doPhysics, backLeftPos * Scale, ref backLeftDistance, dt ) |
419 backRight.Raycast( length - tiltAmount + leanAmount, doPhysics, backRightPos * Scale, ref backRightDistance, dt );

119 private void TryUnfreezeAll( Player owner, Vector3 eyePos, Rotation eyeRot, Vector3 eyeDir )
120 {
121 var tr = Trace.Ray( eyePos, eyePos + eyeDir * MaxTargetDistance )
122 .UseHitboxes()
123 .Ignore( owner, false )
124 .HitLayer( CollisionLayer.Debris )
125 .Run();
126
127 if ( !tr.Hit || !tr.Entity.IsValid() || tr.Entity.IsWorld ) return;
128
129 var rootEnt = tr.Entity.Root;
130 if ( !rootEnt.IsValid() ) return;
131
132 var physicsGroup = rootEnt.PhysicsGroup;
133 if ( physicsGroup == null ) return;
134
135 bool unfrozen = false;
136
137 for ( int i = 0; i < physicsGroup.BodyCount; ++i )
138 {
139 var body = physicsGroup.GetBody( i );
140 if ( !body.IsValid() ) continue;
141
142 if ( body.BodyType == PhysicsBodyType.Static )
143 {
144 body.BodyType = PhysicsBodyType.Dynamic;
145 unfrozen = true;
146 }
147 }
148
149 if ( unfrozen )

314 private void GrabInit( PhysicsBody body, Vector3 startPos, Vector3 grabPos, Rotation rot )
315 {
316 if ( !body.IsValid() )
317 return;
318
319 GrabEnd();
320
321 grabbing = true;
322 heldBody = body;
323 holdDistance = Vector3.DistanceBetween( startPos, grabPos );
324 holdDistance = holdDistance.Clamp( MinTargetDistance, MaxTargetDistance );
325 heldPos = heldBody.Transform.PointToLocal( grabPos );
326 heldRot = rot.Inverse * heldBody.Rotation;
327
328 holdBody.Position = grabPos;
329 holdBody.Rotation = heldBody.Rotation;
330
331 heldBody.Wake();
332 heldBody.EnableAutoSleeping = false;
333
334 holdJoint = PhysicsJoint.Weld
335 .From( holdBody )
336 .To( heldBody, heldPos )
337 .WithLinearSpring( LinearFrequency, LinearDampingRatio, 0.0f )
338 .WithAngularSpring( 0.0f, 0.0f, 0.0f )
339 .Create();
340 }

45 public override void Simulate( Client client )
46 {
47 if ( Owner is not Player owner ) return;
48
49 if ( !IsServer )
50 return;
51
52 using ( Prediction.Off() )
53 {
54 var eyePos = owner.EyePos;
55 var eyeRot = owner.EyeRot;
56 var eyeDir = owner.EyeRot.Forward;
57
58 if ( HeldBody.IsValid() && HeldBody.PhysicsGroup != null )
59 {
60 if ( holdJoint.IsValid && !holdJoint.IsActive )
61 {
62 GrabEnd();
63 }
64 else if ( Input.Pressed( InputButton.Attack1 ) )
65 {
66 if ( HeldBody.PhysicsGroup.BodyCount > 1 )
67 {
68 // Don't throw ragdolls as hard
69 HeldBody.PhysicsGroup.ApplyImpulse( eyeDir * (ThrowForce * 0.5f), true );
70 HeldBody.PhysicsGroup.ApplyAngularImpulse( Vector3.Random * ThrowForce, true );
71 }
72 else
73 {
74 HeldBody.ApplyImpulse( eyeDir * (HeldBody.Mass * ThrowForce) );
75 HeldBody.ApplyAngularImpulse( Vector3.Random * (HeldBody.Mass * ThrowForce) );
76 }
77
78 GrabEnd();
79 }
80 else if ( Input.Pressed( InputButton.Attack2 ) )
81 {
82 timeSinceDrop = 0;
83
84 GrabEnd();
85 }
86 else
87 {
88 GrabMove( eyePos, eyeDir, eyeRot );
89 }
90
91 return;
92 }
93
94 if ( timeSinceDrop < DropCooldown )
95 return;
96
97 var tr = Trace.Ray( eyePos, eyePos + eyeDir * MaxPullDistance )
98 .UseHitboxes()
99 .Ignore( owner, false )
100 .Radius( 2.0f )
101 .HitLayer( CollisionLayer.Debris )
102 .Run();
103
104 if ( !tr.Hit || !tr.Body.IsValid() || !tr.Entity.IsValid() || tr.Entity.IsWorld )
105 return;
106
107 if ( tr.Entity.PhysicsGroup == null )
108 return;
109
110 var modelEnt = tr.Entity as ModelEntity;
111 if ( !modelEnt.IsValid() )
112 return;
113
114 var body = tr.Body;
115
116 if ( Input.Pressed( InputButton.Attack1 ) )
117 {
118 if ( tr.Distance < MaxPushDistance && !IsBodyGrabbed( body ) )
119 {
120 var pushScale = 1.0f - Math.Clamp( tr.Distance / MaxPushDistance, 0.0f, 1.0f );
121 body.ApplyImpulseAt( tr.EndPos, eyeDir * (body.Mass * (PushForce * pushScale)) );
122 }
123 }
124 else if ( Input.Down( InputButton.Attack2 ) )
125 {
126 var physicsGroup = tr.Entity.PhysicsGroup;
127
128 if ( physicsGroup.BodyCount > 1 )
129 {
130 body = modelEnt.PhysicsBody;
131 if ( !body.IsValid() )
132 return;
133 }
134
135 var attachPos = body.FindClosestPoint( eyePos );
136
137 if ( eyePos.Distance( attachPos ) <= AttachDistance )
138 {
139 var holdDistance = HoldDistance + attachPos.Distance( body.MassCenter );
140 GrabStart( modelEnt, body, eyePos + eyeDir * holdDistance, eyeRot );
141 }
142 else if ( !IsBodyGrabbed( body ) )
143 {
144 physicsGroup.ApplyImpulse( eyeDir * -PullForce, true );
145 }
146 }
147 }

212 private void GrabStart( ModelEntity entity, PhysicsBody body, Vector3 grabPos, Rotation grabRot )
213 {
214 if ( !body.IsValid() )
215 return;
216
217 if ( body.PhysicsGroup == null )
218 return;
219
220 if ( IsBodyGrabbed( body ) )
221 return;
222
223 GrabEnd();
224
225 HeldBody = body;
226 HeldRot = grabRot.Inverse * HeldBody.Rotation;
227
228 holdBody.Position = grabPos;
229 holdBody.Rotation = HeldBody.Rotation;
230
231 HeldBody.Wake();
232 HeldBody.EnableAutoSleeping = false;
233
234 collisionJoint = PhysicsJoint.Generic
235 .From( (Owner as Player).PhysicsBody )
236 .To( HeldBody )
237 .Create();
238
239 holdJoint = PhysicsJoint.Weld
240 .From( holdBody )
241 .To( HeldBody, HeldBody.LocalMassCenter )
242 .WithLinearSpring( LinearFrequency, LinearDampingRatio, 0.0f )
243 .WithAngularSpring( AngularFrequency, AngularDampingRatio, 0.0f )
244 .Breakable( HeldBody.Mass * BreakLinearForce, 0 )
245 .Create();
246
247 HeldEntity = entity;
248
249 var client = GetClientOwner();
250 client?.Pvs.Add( HeldEntity );
251 }

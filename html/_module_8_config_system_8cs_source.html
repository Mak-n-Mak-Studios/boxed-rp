137 private static string TypeToString( Type type )
138 {
139 type = GetArrayElementType( type, out int deep );
140 StringBuilder arrayOfText = new( 25 + deep * 2 ); // deep * 2 = how deep the array is * "[]" length. Add 25 for the length of the type string in the array.
141
142 if ( type == typeof( bool ) )
143 {
144 arrayOfText.Append( LocalizationModule.Locale.BaseTrueFalse );
145 }
146 else if ( type == typeof( uint ) || type == typeof( int ) || type == typeof( ushort ) || type == typeof( short ) ||
147 type == typeof( ulong ) || type == typeof( long ) || type == typeof( byte ) || type == typeof( sbyte ) )
148 {
149 arrayOfText.Append( LocalizationModule.Locale.BaseInteger );
150 }
151 else if ( type == typeof( float ) || type == typeof( double ) || type == typeof( decimal ) )
152 {
153 arrayOfText.Append( LocalizationModule.Locale.BaseIntegerDecimal );
154 }
155 else if ( type == typeof( char ) )
156 {
157 arrayOfText.Append( LocalizationModule.Locale.BaseCharacter );
158 }
159 else if ( type == typeof( string ) )
160 {
161 arrayOfText.Append( LocalizationModule.Locale.BaseText );
162 }
163 else
164 {
165 return null;
166 }
167
168 return arrayOfText.Insert( arrayOfText.Length, "[]", deep ).ToString();
169 }

200 protected virtual StringBuilder AppendConfigOption<U>( StringBuilder configDocBuilder, PropertyAttribute configOption, U enclosingObject, int tabsIn )
201 {
202 if ( configOption is not GameConfigOptionInfoAttribute propertyInfo )
203 {
204 return null;
205 }
206
207 Type baseType = GetArrayElementType( propertyInfo.PropertyType, out int deep );
208 string propertyPrimitiveTypeString = TypeToString( propertyInfo.PropertyType );
209 string propertyCustomTypeString = null;
210 string propertyCustomBaseTypeString = null;
211 object propertyDefaultValue = propertyInfo.GetValue<object>( enclosingObject );
212 string spaces = new( ' ', tabsIn * 4 );
213
214 if ( propertyPrimitiveTypeString == null )
215 {
216 string customTypeString = propertyInfo.PropertyType.ToString();
217
218 propertyCustomTypeString = customTypeString[ ( customTypeString.LastIndexOf( '.' ) + 1 ).. ];
219
220 if ( deep > 0 )
221 {
222 string customBaseTypeString = baseType.ToString();
223 propertyCustomBaseTypeString = customBaseTypeString[ ( customBaseTypeString.LastIndexOf( '.' ) + 1 ).. ];
224 }
225 else
226 {
227 propertyCustomBaseTypeString = propertyCustomTypeString;
228 }
229 }
230
231 try
232 {
233 configDocBuilder.Append( spaces )
234 .Append( propertyInfo.Name )
235 .Append( " - " )
236 .Append( LocalizationModule.Locale.ColonPlacement.Format( propertyPrimitiveTypeString ?? propertyCustomTypeString ) )
237 .Append( "\n\n" )
238 .Append( spaces )
239 .Append( LocalizationModule.Locale.ColonPlacement.Format( LocalizationModule.Locale.BaseDescription ) )
240 .Append( '\n' )
241 .Append( spaces )
242 .Append( localizedConfigOptions[ propertyInfo ].Replace( "\n", "\n" + spaces ) )
243 .Append( "\n\n" )
244 .Append( spaces )
245 .Append( LocalizationModule.Locale.ColonPlacement.Format( LocalizationModule.Locale.BaseDefaultValueOfProperty.Format( propertyInfo.Name ) ) )
246 .Append( '\n' )
247 .Append( spaces )
248 .Append( JsonSerializer.Serialize( propertyDefaultValue, propertyInfo.PropertyType, serializerOptions ).Replace( "\n", "\n" + spaces ) );
249 }
250 catch ( KeyNotFoundException )
251 {
252 throw new InvalidOperationException( "The property name given on the [ConfigLocalizedProperty] attribute on the property " +
253 $"{propertyInfo.Name} in {enclosingObject.GetType().FullName} isn't valid. Check that the property name " +
254 "has been typed properly and that the property that the property name refers to has a subclass of the " +
255 "[LocalizationDataProperty] attached to it." );
256 }
257
258 if ( propertyCustomTypeString != null )
259 {
260 configDocBuilder.Append( "\n\n" )
261 .Append( spaces )
262 .Append( "-----------------------------------------------------------------\n\n" )
263 .Append( spaces );
264
265 if ( baseType != typeof( LocaleStrings ) )
266 {
267 configDocBuilder.Append( LocalizationModule.Locale.ColonPlacement.Format( LocalizationModule.Locale.BaseLayoutOfType.Format( propertyCustomBaseTypeString ) ) )
268 .Append( '\n' );
269
270 if ( deep == 0 )
271 {
272 AppendConfigObject( configDocBuilder, propertyDefaultValue, tabsIn + 1, propertyInfo.Name );
273 }
274 else
275 {
276 AppendConfigObject( configDocBuilder, baseType, tabsIn + 1 );
277 }
278 }
279 else // Handle LocaleStrings specially
280 {
281 configDocBuilder.Append( LocalizationModule.Locale.BaseLocaleStringsDescription );
282 }
283 }
284
285 return configDocBuilder;
286 }

312 protected virtual StringBuilder AppendConfigObject<U>( StringBuilder docBuilder, U obj, int tabsIn, string propertyName = null )
313 {
314 if ( obj == null )
315 {
316 throw new ArgumentNullException( nameof( obj ), "Custom config object properties cannot be null. Property name: " + propertyName );
317 }
318
319 Type type = obj.GetType();
320 string spaces = new( ' ', tabsIn * 4 );
321
322 if ( type.IsEnum )
323 {
324 string[] enumNames = type.GetEnumNames();
325
326 if ( enumNames.Length == 0 )
327 {
328 string typeString = type.ToString();
329
330 return docBuilder.Append( spaces )
331 .Append( LocalizationModule.Locale.BaseTypeContainsNothing.Format( typeString[ ( typeString.LastIndexOf( '.' ) + 1 ).. ] ) );
332 }
333
334 string lastEnumConstant = enumNames[ ^1 ];
335
336 foreach ( string enumName in type.GetEnumNames() )
337 {
338 docBuilder.Append( spaces )
339 .Append( enumName );
340
341 if ( enumName != lastEnumConstant )
342 {
343 docBuilder.Append( '\n' );
344 }
345 }
346
347 return docBuilder;
348 }
349
350 IReadOnlyList<PropertyAttribute> configStoreProperties = Library.GetAttribute( type )?.Properties ??
351 throw new ArgumentException( $"The config object of type {type} used in this module's config store does not have the GameConfigObject attribute on it.", nameof( obj ) );
352
353 if ( configStoreProperties.Count == 0 )
354 {
355 string typeString = type.ToString();
356
357 return docBuilder.Append( spaces )
358 .Append( LocalizationModule.Locale.BaseTypeContainsNothing.Format( typeString[ ( typeString.LastIndexOf( '.' ) + 1 ).. ] ) );
359 }
360
361 PropertyAttribute lastProperty = configStoreProperties[ ^1 ];
362
363 foreach ( PropertyAttribute property in configStoreProperties )
364 {
365 try
366 {
367 property.SetValue( obj, property.GetValue<object>( obj ) );
368 }
369 catch ( ArgumentException )
370 {
371 throw new ArgumentException( $"The {property.Name} property within {type} is not both publicly gettable and settable.", nameof( obj ) );
372 }
373
374 if ( property.Attributes.Where( ( attr ) => attr.GetType() == typeof( JsonIgnoreAttribute ) ).Any() )
375 {
376 throw new ArgumentException( $"The {property.Name} property within {type} has [JsonIgnore] on it. This attribute is not compatible with [GameConfigOptionInfo].", nameof( obj ) );
377 }
378
379 bool isConfigOption = AppendConfigOption( docBuilder, property, obj, tabsIn ) != null;
380
381 if ( isConfigOption && lastProperty != property )
382 {
383 docBuilder.Append( "\n\n" )
384 .Append( spaces )
385 .Append( "=================================================================\n\n" );
386 }
387 }
388
389 return docBuilder;
390 }

399 {
400 if ( configDocumentation == null )

404

406 .Append( LocalizationModule.Locale.BaseBeginningOfTypeDocumentation.Format( typeName ) )
407 .Append( @" \\\\\\\\\\\\" )

409
